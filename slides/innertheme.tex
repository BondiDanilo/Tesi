% Here we are using the rounded theme for the overall “feel” of the presentation.
% You can change the specific details by editing that particular option as we have done below.
\useinnertheme{rounded}

% Instead of rounded circles, we will use triangles as the indicator for items.
\setbeamertemplate{itemize items}[triangle]

% The default option for enumerate environment removes the circles
% around the numbers provided by the “rounded” inner theme. Just simple numbers remain.
\setbeamertemplate{enumerate items}[default]
